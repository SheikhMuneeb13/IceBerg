\item{\textit{Exercise:2.20:}}\\

$

i)\\

 Show that the matrix
 
 \[
 A=
 \begin{pmatrix}
 1 & -1 & 5\\
 -1 & 2 & 1\\
 5 & 1 & 3
 \end{pmatrix}
 \] is a symmetric matrix.
 $
 
 \item{\textbf{Solution:}}\\
 
 We've given matrix
 \[
 A=
 \begin{pmatrix}
 1 & -1 & 5\\
 -1 & 2 & 1\\
 5 & 1 & 3
 \end{pmatrix}
 \]
\\

For symmetric matrix\\
$A=A'$

Now,\\

\[
 A'=
 \begin{pmatrix}
 1 & -1 & 5\\
 -1 & 2 & 1\\
 5 & 1 & 3
 \end{pmatrix}
 \]
 
 Clearly, $A=A'$\\
 
Therefore, A is symmetric\\
 
 
$

ii)\\

 Show that the matrix
 
 \[
 A=
 \begin{pmatrix}
 0 & 1 & -1\\
 -1 & 0 & 1\\
 1 & -1 & 0
 \end{pmatrix}
 \] is a skew-symmetric matrix.
 $
 
 \item{\textbf{Solution:}}\\
 
 We've given matrix
 \[
 A=
 \begin{pmatrix}
 0 & 1 & -1\\
 -1 & 0 & 1\\
 1 & -1 & 0
 \end{pmatrix}
 \]
\\

For skew-symmetric matrix\\
$A=-A'$

Now,\\

\[
 A'=
 \begin{pmatrix}
 0 & -1 & 1\\
 1 & 0 & -1\\
 -1 & 1 & 0
 \end{pmatrix}
 \]
 
 Clearly, $A=-A'$\\
 
Therefore, A is skew-symmetric\\

 \item{\textit{Exercise:2.22:}}\\
 
Find  $1/2(A+A')$ and $1/2(A-A')$ when,\\
 
 \[
 A=
 \begin{pmatrix}
 0 & a & b\\
 -a & 0 & c\\
 -b & -c & 0
 \end{pmatrix}
 \]
 \\
 \item{\textbf{Solution:}}\\

We've given matrix,\\

 
 \[
 A=
 \begin{pmatrix}
 0 & a & b\\
 -a & 0 & c\\
 -b & -c & 0
 \end{pmatrix}
 \]\\
 
 Now,\\
 
  \[
 A'=
 \begin{pmatrix}
 0 & -a & -b\\
 a & 0 & -c\\
 b & c & 0
 \end{pmatrix}
 \]
 \\

Therefore,\\

 \[
 A+A'=
 \begin{pmatrix}
 0 & a & b\\
 -a & 0 & c\\
 -b & -c & 0
 \end{pmatrix}
 +
  \begin{pmatrix}
 0 & -a & -b\\
 a & 0 & -c\\
 b & c & 0
 \end{pmatrix}
 \]
\\

\[
 A+A'=
 \begin{pmatrix}
 0 & 0 & 0\\
 0 & 0 & 0\\
 0 & 0 & 0
 \end{pmatrix}
\]\\

Or,\\

$

1/2(A+A')=3x3 Null Matrix
$\\

Also,\\

 \[
 A-A'=
 \begin{pmatrix}
 0 & a & b\\
 -a & 0 & c\\
 -b & -c & 0
 \end{pmatrix}
 -
  \begin{pmatrix}
 0 & -a & -b\\
 a & 0 & -c\\
 b & c & 0
 \end{pmatrix}
 \]
 
 
  \[
 A-A'=
 \begin{pmatrix}
 0 & 2a & 2b\\
 -2a & 0 & 2c\\
 -2b & -2c & 0
 \end{pmatrix}
\]

Now,\\

\[
 1/2(A-A')=
 \begin{pmatrix}
 0 & a & b\\
 -a & 0 & c\\
 -b & -c & 0
 \end{pmatrix}
 =
 A
\]\\
